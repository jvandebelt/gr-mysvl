########################################################################
# Setup library
########################################################################
include(GrPlatform) #define LIB_SUFFIX

include_directories(${Boost_INCLUDE_DIR})
link_directories(${Boost_LIBRARY_DIRS})
list(APPEND mysvl_sources
    weave_impl.cc
    svl_impl.cc
	spectrum_map.cc
	read_fft_params.cc
	hypervisor.cc
	stream.cc
    triggered_demux_impl.cc
    corr_freq_offset_impl.cc
    stream_demux_impl.cc
    sync_channels_impl.cc
    drop_packet_impl.cc )

set(mysvl_sources "${mysvl_sources}" PARENT_SCOPE)
if(NOT mysvl_sources)
	MESSAGE(STATUS "No C++ sources... skipping lib/")
	return()
endif(NOT mysvl_sources)

add_library(gnuradio-mysvl SHARED ${mysvl_sources})
target_link_libraries(gnuradio-mysvl ${Boost_LIBRARIES} ${GNURADIO_ALL_LIBRARIES})
set_target_properties(gnuradio-mysvl PROPERTIES DEFINE_SYMBOL "gnuradio_mysvl_EXPORTS")

if(APPLE)
    set_target_properties(gnuradio-mysvl PROPERTIES
        INSTALL_NAME_DIR "${CMAKE_INSTALL_PREFIX}/lib"
    )
endif(APPLE)

########################################################################
# Install built library files
########################################################################
include(GrMiscUtils)
GR_LIBRARY_FOO(gnuradio-mysvl RUNTIME_COMPONENT "mysvl_runtime" DEVEL_COMPONENT "mysvl_devel")

########################################################################
# Build and register unit test
########################################################################
include(GrTest)

include_directories(${CPPUNIT_INCLUDE_DIRS})

list(APPEND test_mysvl_sources
    ${CMAKE_CURRENT_SOURCE_DIR}/test_mysvl.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_mysvl.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_svl.cc
    ${CMAKE_CURRENT_SOURCE_DIR}/qa_weave.cc
)

add_executable(test-mysvl ${test_mysvl_sources})

target_link_libraries(
  test-mysvl
  ${GNURADIO_RUNTIME_LIBRARIES}
  ${Boost_LIBRARIES}
  ${CPPUNIT_LIBRARIES}
  gnuradio-mysvl
)

GR_ADD_TEST(test_mysvl test-mysvl)

########################################################################
# Print summary
########################################################################
message(STATUS "Using install prefix: ${CMAKE_INSTALL_PREFIX}")
message(STATUS "Building for version: ${VERSION} / ${LIBVER}")
